async function generateLoadstring() { const webhook = document.getElementById('webhook').value; const gameId = document.getElementById('gameId').value || '142823291'; const interval = document.getElementById('interval').value; if (!webhook || !webhook.includes('discord.com/api/webhooks/')) { alert('Please enter a valid Discord webhook URL'); return; } const btn = document.getElementById('generateBtn'); const resultDiv = document.getElementById('result'); const statusDiv = document.getElementById('statusMsg'); btn.disabled = true; btn.textContent = '🔧 Generating...'; resultDiv.style.display = 'block'; statusDiv.innerHTML = '
⏳ Creating script...
'; try { const response = await fetch('/api/generate', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ webhook, gameId, interval }) }); const data = await response.json(); if (data.success) { document.getElementById('loadstringOutput').textContent = data.loadstring; if (data.note) { statusDiv.innerHTML = `
✅ ${data.note}
`; } else { statusDiv.innerHTML = '
✅ Script created! Copy the loadstring above.
'; } } else { statusDiv.innerHTML = `
❌ Error: ${data.error}
`; } } catch (err) { statusDiv.innerHTML = `
❌ Failed: ${err.message}. Check console for details.
`; console.error(err); } finally { btn.disabled = false; btn.textContent = '✨ Generate Loadstring'; } }
